multilevelmenu

a jQuery plugin

JS

	$(function(){
		$('#demo').multilevelmenu();
	})
	---------------------------------------------- or ----------------------------------------------
	$(function(){
		$('#demo').multilevelmenu({
			liH:'22px',                      //菜单项高度
			liBgColor:'#EAEED9',             //菜单项颜色
			liHoverBgColor:'#DFE7CD',        //菜单项鼠标响应颜色
			menuBgColor:'#fafafa',           //菜单区域背景颜色
			menuWidth:960,                   //菜单区域宽度
			eventType:'click',               //触发菜单展开的事件类型“click”or“mouseover”
			fontSize:'14px',                 //字号
			fontFamily:'microsoft YaHei',    //字体
			dataUrl:'menuJsonFile4.json'     //菜单Json数据文件地址
		})
	})
